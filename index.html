    function genQuestion(table) {

    function genAnswers(correct) {

      const opts = new Set([correct]);

      while (opts.size < 4) {

        let off = correct + (Math.floor(Math.random() * 21) - 10);

        if (off > 0 && off !== correct) opts.add(off);

      }

      return [...opts].sort(() => Math.random() - .5);

    }

        btn.onclick = () => startGame('single', i);

        g.appendChild(btn);

      }

    })();


    // ===== GAME ENGINE =====

    function startGame(mode, table) {

      S.mode = mode; S.table = table; S.qi = 0; S.score = 0; S.streak = 0; S.bestStreak = 0; S.correct = 0; S.wrong = 0;

      S.boostActive = false; S.boostMultiplier = 1;

      S.timerOn = mode === 'challenge';

      const count = 10;

      S.questions = [];

      for (let i = 0; i < count; i++) S.questions.push(genQuestion(mode === 'single' ? table : 0));

      $('q-total').textContent = count;

      if (S.timerOn) $('timer-container').classList.remove('hidden');

      else $('timer-container').classList.add('hidden');

      showScreen('screen-game');

      showQuestion();

    }


    function startMixed() { startGame('mixed', 0); }

    function startChallenge() { startGame('challenge', 0); }


    function showQuestion() {

      if (S.qi >= S.questions.length) return endGame();

      const q = S.questions[S.qi];

      $('q-num').textContent = S.qi + 1;

      $('q-text').textContent = `${q.a} × ${q.b} = ?`;

      $('progress-bar').style.width = ((S.qi / S.questions.length) * 100) + '%';

      $('game-score').textContent = S.score;

      $('game-streak').textContent = S.streak;

      $('game-level').textContent = S.level;

      $('feedback-area').classList.add('hidden');


      const opts = genAnswers(q.answer);

      const grid = $('answers-grid');

      grid.innerHTML = '';

      opts.forEach(o => {

        const btn = document.createElement('button');

        const colors = ['bg-pink-500','bg-blue-500','bg-purple-500','bg-teal-500'];

        btn.className = `ans-btn text-white ${colors[opts.indexOf(o) % 4]} pop-in`;

        btn.style.animationDelay = (opts.indexOf(o) * .07) + 's';

        btn.textContent = o;

        btn.onclick = () => answerPicked(o, btn);

        grid.appendChild(btn);

      });


      if (S.timerOn) startTimer();

    }


    function answerPicked(val, btn) {

      if (S.timerId) { clearInterval(S.timerId); S.timerId = null; }

      const q = S.questions[S.qi];

      const isCorrect = val === q.answer;

      const fb = $('feedback-area');

      fb.classList.remove('hidden');


      document.querySelectorAll('#answers-grid .ans-btn').forEach(b => b.disabled = true);


      if (isCorrect) {

        btn.classList.add('correct');

        S.streak++;

        if (S.streak > S.bestStreak) S.bestStreak = S.streak;

        let pts = 10 * S.level;

        // Streak bonus

        if (S.streak >= 3) { pts += S.streak * 2; S.boostMultiplier = 1 + (S.streak * 0.1); }

        if (S.streak === 5) toast('🍍 Tropische Reeks!', '#f97316');

        if (S.streak === 10) toast('🌺 Onstuitbare Aloha!', '#f472b6');

        S.score += Math.round(pts * S.boostMultiplier);

        S.correct++;

        S.stars++;

        addXP(2);

        fb.innerHTML = `<span class="text-green-600 pop-in">✅ Aloha! +${Math.round(pts * S.boostMultiplier)} punten${S.streak >= 3 ? ' 🔥×' + S.streak : ''}</span>`;

      } else {

        btn.classList.add('wrong');

        S.streak = 0; S.boostMultiplier = 1;

        S.wrong++;

        fb.innerHTML = `<span class="text-red-500 shake" style="display:inline-block">❌ Kokosnoot! Het juiste antwoord is <strong>${q.answer}</strong></span>`;

        $('question-card').classList.add('shake');

        setTimeout(() => $('question-card').classList.remove('shake'), 400);

      }


      setTimeout(() => { S.qi++; showQuestion(); }, isCorrect ? 800 : 1800);

    }


    function startTimer() {

      S.timeLeft = Math.max(4, 12 - S.level);

      const max = S.timeLeft;

      const ring = $('timer-ring');

      const circumference = 106.8;

      $('timer-text').textContent = S.timeLeft;

      ring.style.strokeDashoffset = '0';


      S.timerId = setInterval(() => {

        S.timeLeft -= 0.1;

        if (S.timeLeft <= 0) {

          clearInterval(S.timerId); S.timerId = null;

          S.timeLeft = 0;

          // Time's up - count as wrong

          const q = S.questions[S.qi];

          S.streak = 0; S.wrong++;

          const fb = $('feedback-area');

          fb.classList.remove('hidden');

          fb.innerHTML = `<span class="text-red-500">⏰ Te laat! Antwoord: <strong>${q.answer}</strong></span>`;

          document.querySelectorAll('#answers-grid .ans-btn').forEach(b => b.disabled = true);

          setTimeout(() => { S.qi++; showQuestion(); }, 1500);

        }

        $('timer-text').textContent = Math.ceil(S.timeLeft);

        ring.style.strokeDashoffset = ((1 - S.timeLeft / max) * circumference).toString();

      }, 100);

    }


    function endGame() {

      if (S.timerId) { clearInterval(S.timerId); S.timerId = null; }

      // Check special badges

      if (S.correct === S.questions.length) awardBadge('perfect');

      if (S.mode === 'challenge') awardBadge('speed');

      const newB = checkBadges();


      const pct = S.correct / S.questions.length;

      $('result-emoji').textContent = pct >= .8 ? '🎉' : pct >= .5 ? '🥥' : '🌺';

      $('result-title').textContent = pct >= .8 ? 'Fantastisch!' : pct >= .5 ? 'Goed gedaan!' : 'Blijf oefenen!';

      $('result-subtitle').textContent = pct >= .8 ? 'Je bent een tafelkampioen!' : pct >= .5 ? 'Aloha! Dat ging goed!' : '🌴 Aloha! Oefening baart kunst!';

      $('res-score').textContent = S.score;

      $('res-correct').textContent = S.correct;

      $('res-wrong').textContent = S.wrong;

      $('res-streak').textContent = S.bestStreak;


      const nb = $('new-badges');

      nb.innerHTML = '';

      newB.forEach(b => {

        nb.innerHTML += `<div class="badge-card pop-in mb-2">${b.icon} <strong>${b.name}</strong> vrijgespeeld!</div>`;

      });


      showScreen('screen-results');

    }

      const g = $('badges-grid');

      BADGES_DEF.forEach(b => {

        const earned = S.badges.includes(b.id);

      });

    }

    // ===== BATTLE =====

    function setCpuDiff(d) {

      S.cpuDiff = d;

      document.querySelectorAll('.cpu-diff-btn').forEach(b => {

        b.style.borderColor = parseInt(b.dataset.d) === d ? '' : 'transparent';

        if (parseInt(b.dataset.d) === d) b.classList.add('border-2'); 

      });

    }


    function startBattle() {

      S.bPlayerScore = 0; S.bCpuScore = 0; S.bQi = 0;

      S.bQuestions = [];

      for (let i = 0; i < 10; i++) S.bQuestions.push(genQuestion(0));

      showScreen('screen-battle');

      showBattleQ();

    }


    function showBattleQ() {

      if (S.bQi >= S.bQuestions.length) return endBattle();

      const q = S.bQuestions[S.bQi];

      $('b-q-num').textContent = S.bQi + 1;

      $('b-question').textContent = `${q.a} × ${q.b} = ?`;

      $('b-player-score').textContent = S.bPlayerScore;

      $('b-cpu-score').textContent = S.bCpuScore;

      $('b-feedback').classList.add('hidden');

      updateBattleBars();


      const opts = genAnswers(q.answer);

      const grid = $('b-answers');

      grid.innerHTML = '';

      opts.forEach((o, i) => {

        const btn = document.createElement('button');

        const colors = ['bg-pink-500','bg-blue-500','bg-purple-500','bg-teal-500'];

        btn.className = `ans-btn text-white ${colors[i % 4]} pop-in`;

        btn.textContent = o;

        btn.onclick = () => battleAnswer(o, btn);

        grid.appendChild(btn);

      });


      // CPU answers after delay based on difficulty

      const cpuDelay = S.cpuDiff === 1 ? 3000 + Math.random() * 3000 : S.cpuDiff === 2 ? 1500 + Math.random() * 2000 : 800 + Math.random() * 1200;

      const cpuCorrect = S.cpuDiff === 1 ? Math.random() < .55 : S.cpuDiff === 2 ? Math.random() < .75 : Math.random() < .9;

      S._cpuTimeout = setTimeout(() => {

        if (cpuCorrect) S.bCpuScore++;

        $('b-cpu-score').textContent = S.bCpuScore;

        updateBattleBars();

      }, cpuDelay);

    }


    function battleAnswer(val, btn) {

      if (S._cpuTimeout) clearTimeout(S._cpuTimeout);

      const q = S.bQuestions[S.bQi];

      const fb = $('b-feedback');

      fb.classList.remove('hidden');

      document.querySelectorAll('#b-answers .ans-btn').forEach(b => b.disabled = true);


      if (val === q.answer) {

        btn.classList.add('correct');

        S.bPlayerScore++;

        fb.innerHTML = '<span class="text-green-600">✅ Aloha! Goed!</span>';

        // CPU still gets chance

        const cpuCorrect = S.cpuDiff === 1 ? Math.random() < .4 : S.cpuDiff === 2 ? Math.random() < .65 : Math.random() < .85;

        if (cpuCorrect) S.bCpuScore++;

      } else {

        btn.classList.add('wrong');

        fb.innerHTML = `<span class="text-red-500">❌ Kokosnoot! Antwoord: ${q.answer}</span>`;

        // CPU more likely correct when player wrong

        const cpuCorrect = S.cpuDiff === 1 ? Math.random() < .6 : Math.random() < .8;

        if (cpuCorrect) S.bCpuScore++;

      }

      $('b-player-score').textContent = S.bPlayerScore;

      $('b-cpu-score').textContent = S.bCpuScore;

      updateBattleBars();


      setTimeout(() => { S.bQi++; showBattleQ(); }, 1200);

    }


    function updateBattleBars() {

      const total = S.bPlayerScore + S.bCpuScore || 1;

      $('b-player-bar').style.width = ((S.bPlayerScore / 10) * 100) + '%';

      $('b-cpu-bar').style.width = ((S.bCpuScore / 10) * 100) + '%';

    }


    function endBattle() {

      const won = S.bPlayerScore > S.bCpuScore;

      const tie = S.bPlayerScore === S.bCpuScore;

      $('br-emoji').textContent = won ? '🌴' : tie ? '🥥' : '🌺';

      $('br-title').textContent = won ? 'Jij wint!' : tie ? 'Gelijkspel!' : 'CPU wint!';

      $('br-player').textContent = S.bPlayerScore;

      $('br-cpu').textContent = S.bCpuScore;

      if (won) { awardBadge('battle_win'); S.stars += 5; addXP(5); }

      checkBadges();

      showScreen('screen-battle-result');

    }
